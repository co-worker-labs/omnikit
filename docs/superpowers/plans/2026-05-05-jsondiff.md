> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a new `/jsondiff` tool that performs semantic-level comparison of two JSON objects, detecting added, removed, and updated fields with nested path support.

**Architecture:** Standalone tool with no shared code with the existing Text Diff. Core logic in `libs/jsondiff/` wraps `json-diff-ts` for changeset generation and builds a custom merged tree for side-by-side rendering. Standard page pattern with `page.tsx` + `jsondiff-page.tsx`.

**Tech Stack:** TypeScript, React 19, Next.js 16 App Router, Tailwind CSS 4, json-diff-ts, json5, next-intl, Vitest

```ts
import { diff, atomizeChangeset } from "json-diff-ts";
import type { JsonDiffResult, FlatChange } from "./types";
import { buildMergedTree } from "./tree";

function mergeTypeChanges(flatList: FlatChange[]): FlatChange[] {
  const merged: FlatChange[] = [];
  for (let i = 0; i < flatList.length; i++) {
    const curr = flatList[i];
    const next = flatList[i + 1];
    if (curr.type === "REMOVE" && next && next.type === "ADD" && curr.path === next.path) {
      merged.push({
        ...curr,
        type: "UPDATE",
        value: next.value,
        oldValue: curr.oldValue,
      });
      i++;
    } else {
      merged.push(curr);
    }
  }
  return merged;
}

export function compareJson(left: unknown, right: unknown): JsonDiffResult {
  const changeset = diff(left, right);
  const rawFlat = atomizeChangeset(changeset) as FlatChange[];
  const flatList = mergeTypeChanges(rawFlat);
  const mergedTree = buildMergedTree(left, right, changeset);

  let updates = 0;
  let adds = 0;
  let removes = 0;
  for (const item of flatList) {
    if (item.type === "UPDATE") updates++;
    else if (item.type === "ADD") adds++;
    else if (item.type === "REMOVE") removes++;
  }

  return { changeset, flatList, mergedTree, stats: { updates, adds, removes } };
}
```

```ts
import type { MergedTreeNode, DiffType } from "./types";

interface ChangeEntry {
  type: string;
  key: string;
  value?: unknown;
  oldValue?: unknown;
  changes?: ChangeEntry[];
}

function isObject(val: unknown): val is Record<string, unknown> {
  return typeof val === "object" && val !== null && !Array.isArray(val);
}

function buildChangeMap(changeset: ChangeEntry[]): Map<string, ChangeEntry[]> {
  const map = new Map<string, ChangeEntry[]>();
  for (const c of changeset) {
    const existing = map.get(c.key);
    if (existing) existing.push(c);
    else map.set(c.key, [c]);
  }
  return map;
}

function resolveType(entries: ChangeEntry[]): DiffType {
  if (entries.length === 1) return entries[0].type as DiffType;
  const hasRemove = entries.some((e) => e.type === "REMOVE");
  const hasAdd = entries.some((e) => e.type === "ADD");
  if (hasRemove && hasAdd) return "UPDATE";
  return entries[0].type as DiffType;
}

export function buildMergedTree(
  left: unknown,
  right: unknown,
  changeset: ChangeEntry[]
): MergedTreeNode {
  if (isObject(left) && isObject(right)) {
    return {
      key: "$",
      type: "UNCHANGED",
      children: mergeObjectChildren(left, right, changeset),
    };
  }
  if (Array.isArray(left) && Array.isArray(right)) {
    return {
      key: "$",
      type: "UNCHANGED",
      children: mergeArrayChildren(left, right, changeset),
    };
  }
  return { key: "$", type: "UNCHANGED", leftValue: left, rightValue: right };
}

function mergeObjectChildren(
  left: Record<string, unknown>,
  right: Record<string, unknown>,
  changeset: ChangeEntry[]
): MergedTreeNode[] {
  const changeMap = buildChangeMap(changeset);
  const seen = new Set<string>();
  const nodes: MergedTreeNode[] = [];

  for (const key of Object.keys(left)) {
    seen.add(key);
    const changes = changeMap.get(key);
    const inRight = key in right;
    nodes.push(buildNode(key, left[key], right[key], inRight, true, changes));
  }

  for (const key of Object.keys(right)) {
    if (!seen.has(key)) {
      const changes = changeMap.get(key);
      nodes.push(buildNode(key, undefined, right[key], true, false, changes));
    }
  }

  return nodes;
}

function mergeArrayChildren(
  left: unknown[],
  right: unknown[],
  changeset: ChangeEntry[]
): MergedTreeNode[] {
  const maxLen = Math.max(left.length, right.length);
  const changeMap = buildChangeMap(changeset);
  const nodes: MergedTreeNode[] = [];

  for (let i = 0; i < maxLen; i++) {
    const key = String(i);
    const leftVal = i < left.length ? left[i] : undefined;
    const rightVal = i < right.length ? right[i] : undefined;
    const inLeft = i < left.length;
    const inRight = i < right.length;
    nodes.push(buildNode(key, leftVal, rightVal, inRight, inLeft, changeMap.get(key)));
  }

  return nodes;
}

function buildNode(
  key: string,
  leftVal: unknown,
  rightVal: unknown,
  inRight: boolean,
  inLeft: boolean,
  changes: ChangeEntry[] | undefined
): MergedTreeNode {
  const type: DiffType = !changes || changes.length === 0 ? "UNCHANGED" : resolveType(changes);

  if (type === "ADD") {
    if (isObject(rightVal)) {
      return { key, type, children: mergeObjectChildren({}, rightVal, []) };
    }
    if (Array.isArray(rightVal)) {
      return { key, type, children: mergeArrayChildren([], rightVal, []) };
    }
    return { key, type, rightValue: rightVal };
  }

  if (type === "REMOVE") {
    if (isObject(leftVal)) {
      return { key, type, children: mergeObjectChildren(leftVal, {}, []) };
    }
    if (Array.isArray(leftVal)) {
      return { key, type, children: mergeArrayChildren(leftVal, [], []) };
    }
    return { key, type, leftValue: leftVal };
  }

  const nestedChanges = (changes || []).flatMap((c) => c.changes || []);

  if (isObject(leftVal) && isObject(rightVal)) {
    return {
      key,
      type,
      children: mergeObjectChildren(leftVal, rightVal, nestedChanges),
    };
  }

  if (Array.isArray(leftVal) && Array.isArray(rightVal)) {
    return {
      key,
      type,
      children: mergeArrayChildren(leftVal, rightVal, nestedChanges),
    };
  }

  return {
    key,
    type,
    leftValue: inLeft ? leftVal : undefined,
    rightValue: inRight ? rightVal : undefined,
  };
}
```

```tsx
"use client";

import { useEffect, useRef, useState, type DragEvent, type ChangeEvent } from "react";
import json5 from "json5";
import { ArrowLeftRight, Copy, Upload, X } from "lucide-react";
import { useTranslations } from "next-intl";
import Layout from "../../../components/layout";
import { Button } from "../../../components/ui/button";
import { Badge } from "../../../components/ui/badge";
import { NeonTabs } from "../../../components/ui/tabs";
import { showToast } from "../../../libs/toast";
import { STORAGE_KEYS } from "../../../libs/storage-keys";
import { useIsMobile } from "../../../hooks/use-is-mobile";
import { compareJson } from "../../../libs/jsondiff/compare";
import { isBinaryFile } from "../../../libs/file/binary-sniff";
import { MAX_FILE_BYTES } from "../../../libs/file/limits";
import type { JsonDiffResult, ViewMode } from "../../../libs/jsondiff/types";
import type { MergedTreeNode } from "../../../libs/jsondiff/types";

const DEBOUNCE_MS = 300;
const SIZE_WARNING = 1024 * 1024;

type Persisted = { viewMode: ViewMode };
const DEFAULT_PERSISTED: Persisted = { viewMode: "side" };

function readPersisted(): Persisted {
  if (typeof window === "undefined") return DEFAULT_PERSISTED;
  const raw = window.localStorage.getItem(STORAGE_KEYS.jsondiff);
  if (!raw) return DEFAULT_PERSISTED;
  try {
    const p = JSON.parse(raw) as Partial<Persisted>;
    return { viewMode: p.viewMode === "summary" ? "summary" : "side" };
  } catch {
    return DEFAULT_PERSISTED;
  }
}

function tryParse(input: string): { value: unknown } | { error: string } {
  const trimmed = input.trim();
  if (!trimmed) return { error: "Empty input" };
  try {
    return { value: JSON.parse(trimmed) };
  } catch {
    try {
      return { value: json5.parse(trimmed) };
    } catch (e) {
      return { error: e instanceof SyntaxError ? e.message : String(e) };
    }
  }
}

function formatJson(text: string): string | null {
  try {
    return JSON.stringify(JSON.parse(text), null, 2);
  } catch {
    try {
      return JSON.stringify(json5.parse(text), null, 2);
    } catch {
      return null;
    }
  }
}

type JsonValue = string | number | boolean | null | undefined;

function formatValue(val: unknown): string {
  if (val === undefined) return "";
  if (val === null) return "null";
  if (typeof val === "string") return `"${val}"`;
  if (typeof val === "number" || typeof val === "boolean") return String(val);
  if (Array.isArray(val)) return `[${val.length} items]`;
  if (typeof val === "object") return `{${Object.keys(val).length} keys}`;
  return String(val);
}

function typeBgClass(type: string): string {
  switch (type) {
    case "UPDATE":
      return "bg-amber-500/10";
    case "ADD":
      return "bg-green-500/10";
    case "REMOVE":
      return "bg-red-500/10";
    default:
      return "";
  }
}

function typeTextClass(type: string): string {
  switch (type) {
    case "UPDATE":
      return "text-amber-500";
    case "ADD":
      return "text-green-500";
    case "REMOVE":
      return "text-red-500";
    default:
      return "text-fg-primary";
  }
}

function badgeVariant(type: string): "default" | "cyan" | "purple" | "danger" {
  switch (type) {
    case "UPDATE":
      return "purple";
    case "ADD":
      return "cyan";
    case "REMOVE":
      return "danger";
    default:
      return "default";
  }
}

function TreeNodeRow({
  node,
  depth,
  side,
}: {
  node: MergedTreeNode;
  depth: number;
  side: "left" | "right";
}) {
  const [expanded, setExpanded] = useState(true);
  const hasChildren = node.children && node.children.length > 0;
  const val = side === "left" ? node.leftValue : node.rightValue;
  const showValue = side === "left" ? node.type !== "ADD" : node.type !== "REMOVE";
  const dimmed =
    (side === "left" && node.type === "ADD") || (side === "right" && node.type === "REMOVE");

  return (
    <>
      <div
        className={`flex items-center font-mono text-sm ${typeBgClass(node.type)} ${dimmed ? "opacity-30" : ""}`}
        style={{ paddingLeft: depth * 20 }}
      >
        {hasChildren ? (
          <button
            type="button"
            onClick={() => setExpanded(!expanded)}
            className="w-4 text-fg-muted cursor-pointer shrink-0"
          >
            {expanded ? "▾" : "▸"}
          </button>
        ) : (
          <span className="w-4 shrink-0" />
        )}
        <span className={typeTextClass(node.type)}>
          {node.key}
          {!hasChildren && showValue && (
            <span className="text-fg-secondary">: {formatValue(val)}</span>
          )}
          {!hasChildren && !showValue && <span className="text-fg-secondary">: —</span>}
          {hasChildren && (
            <span className="text-fg-muted">
              {Array.isArray(
                (side === "left" ? node.leftValue : node.rightValue) ||
                  (side === "left" ? node.rightValue : node.leftValue)
              )
                ? " ["
                : " {"}
            </span>
          )}
        </span>
      </div>
      {hasChildren &&
        expanded &&
        node.children!.map((child, i) => (
          <TreeNodeRow key={`${child.key}-${i}`} node={child} depth={depth + 1} side={side} />
        ))}
      {hasChildren && expanded && (
        <div className="font-mono text-sm text-fg-muted" style={{ paddingLeft: depth * 20 }}>
          <span className="w-4" />
          {Array.isArray(
            (side === "left" ? node.leftValue : node.rightValue) ||
              (side === "left" ? node.rightValue : node.leftValue)
          )
            ? "]"
            : "}"}
        </div>
      )}
    </>
  );
}

function SideBySideView({ tree }: { tree: MergedTreeNode }) {
  return (
    <div className="grid grid-cols-1 md:grid-cols-2 gap-0 border border-border-default rounded-lg overflow-hidden text-sm">

        {tree.children?.map((child, i) => (
          <TreeNodeRow key={`l-${child.key}-${i}`} node={child} depth={0} side="left" />
        ))}
      </div>
      <div>
        {tree.children?.map((child, i) => (
          <TreeNodeRow key={`r-${child.key}-${i}`} node={child} depth={0} side="right" />
        ))}
      </div>
    </div>
  );
}

function SummaryView({ result }: { result: JsonDiffResult }) {
  const t = useTranslations("jsondiff");
  if (result.flatList.length === 0) {
    return <div className="text-center py-8 text-fg-secondary">{t("noChanges")}</div>;
  }

  return (
    <div className="space-y-1">
      {result.flatList.map((change, i) => (
        <div
          key={`${change.path}-${i}`}
          className={`flex items-center gap-3 px-3 py-1.5 rounded font-mono text-sm ${typeBgClass(change.type)}`}
        >
          <Badge variant={badgeVariant(change.type)}>{change.type}</Badge>
          <span className="text-fg-secondary">{change.path}</span>
          {change.type === "UPDATE" && (
            <span className="ml-auto text-fg-secondary">
              <span className="line-through">{formatValue(change.oldValue)}</span>
              <span className="mx-1">→</span>
              <span className={typeTextClass(change.type)}>{formatValue(change.value)}</span>
            </span>
          )}
          {change.type === "ADD" && (
            <span className={`ml-auto ${typeTextClass(change.type)}`}>
              {formatValue(change.value)}

function StatsBar({ stats }: { stats: JsonDiffResult["stats"] }) {
  const t = useTranslations("jsondiff");
  const parts: string[] = [];
  if (stats.updates > 0) parts.push(t("stats.updates", { count: stats.updates }));

  if (parts.length === 0) return null;

  return (

  );
}

function DiffInput({
  value,
  onChange,
  label,
  placeholder,
  accent,
  error,
}: {
  value: string;
  onChange: (v: string) => void;
  label: string;
  placeholder: string;
  accent: "cyan" | "purple";
  error: string | null;
}) {
  const t = useTranslations("jsondiff");
  const tc = useTranslations("common");
  const fileInputRef = useRef<HTMLInputElement | null>(null);
  const inputId = useRef(`file-${Math.random().toString(36).slice(2)}`);

  const dotClass = accent === "cyan" ? "bg-accent-cyan/60" : "bg-accent-purple/60";
  const labelClass = accent === "cyan" ? "text-accent-cyan" : "text-accent-purple";

  async function handleFile(file: File) {
    if (file.size > MAX_FILE_BYTES) {
      showToast(t("largeWarning"), "danger", 3000);
      return;
    }
    const binary = await isBinaryFile(file);
    if (binary) return;
    const text = await file.text();
    onChange(text);
  }

  function onDrop(ev: DragEvent<HTMLDivElement>) {
    ev.preventDefault();
    const file = ev.dataTransfer.files?.[0];
    if (file) void handleFile(file);
  }

  return (
    <div className="flex flex-col">
      <div className="flex flex-wrap justify-between items-center">
        <div className="flex items-center gap-2">
          <span className={`w-2.5 h-2.5 rounded-full ${dotClass}`} />
          <span className={`font-mono text-sm font-semibold ${labelClass}`}>{label}</span>
        </div>
        <div className="flex items-center gap-3">
          <button
            type="button"
            onClick={() => fileInputRef.current?.click()}
            className="inline-flex items-center gap-1 text-xs text-fg-secondary hover:text-accent-cyan transition-colors cursor-pointer"
          >
            <Upload size={14} />
            {t("uploadFile")}
          </button>
          <input
            ref={fileInputRef}
            id={inputId.current}
            type="file"
            className="hidden"
            accept=".json,.txt"
            onChange={(e: ChangeEvent<HTMLInputElement>) => {
              const file = e.target.files?.[0];
              e.target.value = "";
              if (file) void handleFile(file);
            }}
          />
          <button
            type="button"
            className="text-danger text-xs hover:text-danger/80 transition-colors cursor-pointer"
            onClick={() => {
              onChange("");
              showToast(tc("cleared"), "danger", 2000);
            }}
          >
            {tc("clear")}
          </button>
        </div>
      </div>
      <div
        onDrop={onDrop}
        onDragOver={(e) => e.preventDefault()}
        className="relative mt-1"
        style={{ height: "30vh" }}
      >
        <textarea
          placeholder={placeholder}
          value={value}
          onChange={(e) => onChange(e.target.value)}
          className="w-full h-full resize-none rounded-lg border border-border-default bg-bg-input text-fg-primary font-mono text-sm p-3 focus:outline-none focus:border-accent-cyan transition-colors"
        />
      </div>
      {error && <p className="mt-1 text-xs text-danger">{error}</p>}
    </div>
  );
}

function DiffBody() {
  const t = useTranslations("jsondiff");
  const tc = useTranslations("common");

  const [leftText, setLeftText] = useState("");
  const [rightText, setRightText] = useState("");
  const [viewMode, setViewMode] = useState<ViewMode>("side");
  const [hydrated, setHydrated] = useState(false);
  const [result, setResult] = useState<JsonDiffResult | null>(null);
  const [leftError, setLeftError] = useState<string | null>(null);
  const [rightError, setRightError] = useState<string | null>(null);

  const isMobile = useIsMobile();

  useEffect(() => {
    const p = readPersisted();
    setViewMode(p.viewMode);
    setHydrated(true);
  }, []);

  useEffect(() => {
    if (!hydrated) return;
    window.localStorage.setItem(STORAGE_KEYS.jsondiff, JSON.stringify({ viewMode }));
  }, [viewMode, hydrated]);

  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    if (!hydrated) return;
    if (timerRef.current) clearTimeout(timerRef.current);

    const bothEmpty = !leftText.trim() && !rightText.trim();
    if (bothEmpty) {
      setResult(null);
      setLeftError(null);
      setRightError(null);
      return;
    }

    timerRef.current = setTimeout(() => {
      setLeftError(null);
      setRightError(null);

      const leftParse = tryParse(leftText);
      const rightParse = tryParse(rightText);

      if ("error" in leftParse) {
        setLeftError(leftParse.error);
        setResult(null);
        if ("error" in rightParse) setRightError(rightParse.error);
        return;
      }
      if ("error" in rightParse) {
        setRightError(rightParse.error);
        setResult(null);
        return;
      }

      const totalSize = leftText.length + rightText.length;
      if (totalSize > SIZE_WARNING) {
        showToast(t("largeWarning"), "warning", 3000);
      }

      const r = compareJson(leftParse.value, rightParse.value);
      setResult(r);

      if (r.stats.updates === 0 && r.stats.adds === 0 && r.stats.removes === 0) {
        showToast(t("noChanges"), "success", 3000);
      }
    }, DEBOUNCE_MS);

    return () => {
      if (timerRef.current) clearTimeout(timerRef.current);
    };
  }, [leftText, rightText, hydrated]);

  function onSwap() {
    const tmp = leftText;
    setLeftText(rightText);
    setRightText(tmp);
  }

  function onFormatBoth() {
    const lf = formatJson(leftText);
    const rf = formatJson(rightText);
    if (lf) setLeftText(lf);
    if (rf) setRightText(rf);
    if (!lf && leftText.trim()) showToast(t("parseError", { message: "left" }), "danger", 3000);
    if (!rf && rightText.trim()) showToast(t("parseError", { message: "right" }), "danger", 3000);
  }

  function onCopyResult() {
    if (!result || result.flatList.length === 0) return;
    const lines = result.flatList.map((c) => {
      if (c.type === "UPDATE")
        return `[UPDATE] ${c.path}: ${JSON.stringify(c.oldValue)} → ${JSON.stringify(c.value)}`;
      if (c.type === "ADD") return `[ADD] ${c.path}: ${JSON.stringify(c.value)}`;
      return `[REMOVE] ${c.path}`;
    });
    void navigator.clipboard.writeText(lines.join("\n"));
    showToast(tc("copied"), "success", 2000);
  }

  function onClearAll() {
    setLeftText("");
    setRightText("");
    setResult(null);
    setLeftError(null);
    setRightError(null);
    showToast(tc("allCleared"), "danger", 2000);
  }

  const hasInput = leftText.trim().length > 0 || rightText.trim().length > 0;
  const effectiveView = isMobile ? "summary" : viewMode;

  return (
    <>
      <div className={isMobile ? "flex flex-col gap-4" : "grid grid-cols-1 md:grid-cols-2 gap-4"}>
        <DiffInput
          value={leftText}
          onChange={setLeftText}
          label={t("left")}
          placeholder={t("leftPlaceholder")}
          accent="cyan"
          error={leftError}
        />
        <DiffInput
          value={rightText}
          onChange={setRightText}
          label={t("right")}
          placeholder={t("rightPlaceholder")}
          accent="purple"
          error={rightError}
        />
      </div>

      <div className="flex items-center gap-3 justify-center my-3">
        <Button
          variant="outline"
          size="sm"
          disabled={!hasInput}
          onClick={onSwap}
          className="rounded-full !gap-1"
        >
          <ArrowLeftRight size={14} />
          {t("swap")}
        </Button>
        <Button
          variant="outline"
          size="sm"
          disabled={!hasInput}
          onClick={onFormatBoth}
          className="rounded-full !gap-1"
        >
          {t("formatBoth")}
        </Button>
        <Button
          variant="outline"
          size="sm"
          disabled={!result || result.flatList.length === 0}
          onClick={onCopyResult}
          className="rounded-full !gap-1"
        >
          <Copy size={14} />
          {t("copyResult")}
        </Button>
        <Button
          variant="danger"
          size="sm"
          disabled={!hasInput}
          onClick={onClearAll}
          className="rounded-full !gap-1"
        >
          <X size={14} />
          {tc("clearAll")}
        </Button>
      </div>

      {result && (
        <>
          {!isMobile && (
            <div className="flex items-center gap-4 mb-3">
              <div
                role="radiogroup"
                className="inline-flex rounded-full border border-border-default overflow-hidden"
              >
                {(["side", "summary"] as const).map((v) => (
                  <button
                    key={v}
                    type="button"
                    role="radio"
                    aria-checked={viewMode === v}
                    onClick={() => setViewMode(v)}
                    className={
                      "px-4 py-1.5 text-sm font-medium transition-colors cursor-pointer " +
                      (viewMode === v
                        ? "bg-accent-cyan text-bg-base"
                        : "bg-transparent text-fg-secondary hover:bg-bg-elevated")
                    }
                  >
                    {v === "side" ? t("view.sideBySide") : t("view.summary")}
                  </button>
                ))}
              </div>
            </div>
          )}

          <div className="border border-border-default rounded-lg p-3 bg-bg-surface">
            {effectiveView === "side" ? (
              <SideBySideView tree={result.mergedTree} />
            ) : (
              <SummaryView result={result} />
            )}
            <StatsBar stats={result.stats} />
          </div>
        </>
      )}
    </>
  );
}
```
